\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
    node distance=0.85cm and 1.2cm,
    >=Latex,
    font=\small,
    every node/.style={align=center},
    startstop/.style={
        ellipse, draw, thick, fill=gray!10,
        minimum width=3.2cm, minimum height=0.9cm
    },
    io/.style={
        trapezium, trapezium left angle=75, trapezium right angle=105,
        draw, thick, fill=green!8,
        minimum height=1.05cm, text width=5.4cm
    },
    process/.style={
        rectangle, rounded corners=3pt, draw, thick, fill=blue!6,
        minimum height=1.15cm, text width=6.9cm
    },
    decision/.style={
        diamond, aspect=2.2, draw, thick, fill=orange!10,
        text width=3.7cm, inner sep=1pt
    },
    line/.style={draw, thick, ->}
]

% nodes
\node (start) [startstop] {开始};

\node (input) [io, below=of start] {
赛题信息输入\\
13 种攻击动作、机器人参数、比赛规则与约束条件
};

\node (feasible) [process, below=of input] {
约束建模与可行域筛选\\
结合动力学约束、平衡约束与赛事规则，剔除不具备物理可行性或实战意义的动作参数组合
};

\node (judge1) [decision, below=1.1cm of feasible] {是否满足\\可行性约束？};

\node (param) [process, below=1.0cm of judge1] {
动作参数化表达\\
构建原始参数矩阵 $X$，提取末端速度、等效质量、命中概率、动作时长、质心偏移等指标
};

\node (mech) [process, below=of param] {
机理分析与评价降维\\
基于“攻击效果—自身稳定—实战适用”三条主线，建立一级指标体系，
将原始多参数压缩为 $A_i,\;S_i,\;U_i$
};

\node (norm) [process, below=of mech] {
指标同向化与无量纲化\\
区分效益型/成本型指标，消除量纲差异，保证不同动作之间具有可比性
};

\node (weight) [process, below=of norm] {
综合评价模型构建\\
设定权重参数与修正系数，将攻击收益、稳定性、实战适用性耦合为综合得分
$\mathrm{Score}_i$
};

\node (rank) [process, below=of weight] {
动作排序与优选集提取\\
对 13 种动作进行综合评分排序，识别高收益且高稳定的优选攻击动作
};

\node (sens) [process, below=of rank] {
敏感性与稳健性分析\\
扰动权重参数、修正系数及关键输入参数，检验排序结果是否稳定可靠
};

\node (judge2) [decision, below=1.1cm of sens] {排序结果\\是否稳定？};

\node (output) [process, below=1.0cm of judge2] {
结果解释与策略输出\\
给出优先采用的攻击动作及强弱顺序，并从攻击收益、稳定性和实战适用性角度解释其成因
};

\node (end) [startstop, below=of output] {结束};

% side node
\node (revise) [process, right=2.8cm of judge2, text width=4.8cm] {
返回修正\\
调整权重设定、修正参数赋值或重新校准一级指标构造
};

\node (reject) [process, right=2.8cm of judge1, text width=4.8cm] {
重新修正参数\\
缩小参数取值范围或重新设定动作可行域
};

% arrows
\path [line] (start) -- (input);
\path [line] (input) -- (feasible);
\path [line] (feasible) -- (judge1);

\path [line] (judge1) -- node[left]{是} (param);
\path [line] (param) -- (mech);
\path [line] (mech) -- (norm);
\path [line] (norm) -- (weight);
\path [line] (weight) -- (rank);
\path [line] (rank) -- (sens);
\path [line] (sens) -- (judge2);

\path [line] (judge2) -- node[left]{是} (output);
\path [line] (output) -- (end);

\path [line] (judge1) -- node[above]{否} (reject);
\path [line] (reject.west) |- ([xshift=0.2cm]feasible.east);

\path [line] (judge2) -- node[above]{否} (revise);
\path [line] (revise.west) |- ([xshift=0.2cm]weight.east);

\end{tikzpicture}
\caption{问题一模型求解流程图}
\label{fig:q1_flowchart}
\end{figure}
